---
title: "List Articles"
api: "GET https://www.pacekb.com/api/v2/articles"
description: "This endpoint gets all articles from your Knowledge Base."
---

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="message" type="string">
  If there's an error an accompanying error message will be visible
</ResponseField>

<ResponseField name="data" type="array">

List of all articles

<Expandable title="Toggle object in array">

<ResponseField name="other_articles" type="array">
  An array of all languages associated with this article in the format of \{language: 'fr'\}
</ResponseField>

<ResponseField name="id" type="string">
  The unique ID of the article
</ResponseField>

<ResponseField name="name" type="string">
  The name of the article
</ResponseField>

<ResponseField name="status" type="string">
  The current article status: (draft, draftCopy, published)
</ResponseField>

<ResponseField name="last_edited_date" type="date">
  Date when the article was last edited
</ResponseField>

<ResponseField name="ai_summary" type="string">
  AI generated summary of the article (if enabled in integrations)
</ResponseField>

<ResponseField name="users_articles_created_byTousers" type="object">
  The user that created the article
  <Expandable title="Users object">
    <ResponseField name="username" type="string">
      Username of user
    </ResponseField>
    <ResponseField name="first_name" type="string">
      First name of user
    </ResponseField>
    <ResponseField name="last_name" type="string">
      Last name of user
    </ResponseField>
    <ResponseField name="email_address" type="string">
      Email address of user
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="_count" type="object">
  Statistics of the article
  <Expandable title="Statistics object">
    <ResponseField name="tracking" type="number">
      Number of article views
    </ResponseField>
  </Expandable>
</ResponseField>

</Expandable>
</ResponseField>

<ResponseField name="timezone" type="string">
  Your timezone preference
</ResponseField>

<ResponseField name="limit" type="number">
  Your limit of articles
</ResponseField>

<ResponseField name="languages" type="array">
  Array of all language codes your Knowledge Base has enabled
</ResponseField>

<ResponseField name="ai_summary" type="boolean">
  AI Summary integration is enabled
</ResponseField>

<RequestExample>

```bash Example Request
curl --location --request GET 'https://www.pacekb.com/api/v2/articles' \
--header 'Content-Type: application/json' \
--header 'Authorization: Token <token>' \
```

</RequestExample>
