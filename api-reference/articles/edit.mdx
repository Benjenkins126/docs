---
title: "Edit Article"
api: "POST https://www.pacekb.com/api/v2/articles/[id]"
description: "This endpoint edits an existing article."
---

<ParamField path="id" type="string">
    ID of the article to edit
</ParamField>

### Body

<ParamField body="title" type="string" required>
    Title of article
</ParamField>

<ParamField body="slug" type="string" required>
    Slug of article
</ParamField>

<ParamField body="categories" type="array" required>
    Array of all category objects
    <Expandable title="Category object">
        <ResponseField name="value" type="string">
            ID of the category
        </ResponseField>
    </Expandable>
</ParamField>

<ParamField body="tags" type="array">
    Array of all tags objects
    <Expandable title="Tag object">
        <ResponseField name="value" type="string">
            ID of the tag
        </ResponseField>
    </Expandable>
</ParamField>

<ParamField body="excerpt" type="string" required>
    Excerpt of article (Max 160 characters)
</ParamField>

<ParamField body="ordering" type="number" required>
    Order priority of article
</ParamField>

<ParamField body="status" type="string" required>
    Status of article: draft or published
</ParamField>

<ParamField body="content" type="string" required>
    Article content (HTML accepted)
</ParamField>

<ParamField body="hideFromSearch" type="boolean" required>
    Hide article from search
</ParamField>

<ParamField body="seoBannerUrl" type="string">
    URL of SEO banner image
</ParamField>

<ParamField body="articleReviewDate" type="date">
    Date of next review of article
</ParamField>

<ParamField body="lang" type="string" required>
    Language code if not an english tag
</ParamField>

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="message" type="string">
  Success/error message
</ResponseField>
