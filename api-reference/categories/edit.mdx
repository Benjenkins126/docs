---
title: "Edit Category"
api: "POST https://www.pacekb.com/api/v2/categories/[id]"
description: "This endpoint edits an existing category."
---

<ParamField path="id" type="string">
    ID of the category to edit
</ParamField>

### Body

<ParamField body="name" type="string" required>
    Name of category
</ParamField>

<ParamField body="slug" type="string" required>
    Slug of category
</ParamField>

<ParamField body="priority" type="number" required>
    Ordering priority of article
</ParamField>

<ParamField body="categoryParent" type="string">
    ID of category parent
</ParamField>

<ParamField body="description" type="string">
    Description of category
</ParamField>

<ParamField body="longDescription" type="string">
    HTML-supported description of category
</ParamField>

<ParamField body="thumbnailUrl" type="string">
    URL of thumbnail image
</ParamField>

<ParamField body="iconUrl" type="string">
    URL of icon image
</ParamField>

<ParamField body="lang" type="string" required>
    Language code if not an english tag
</ParamField>

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="message" type="string">
  Success/error message
</ResponseField>

<RequestExample>

```bash Example Request
curl --location --request POST 'https://www.pacekb.com/api/v2/categories/[id]' \
--header 'Content-Type: application/json' \
--header 'Authorization: Bearer <api_key>' \
--data-raw '{
    "name": "Getting Started",
    "slug": "getting-started",
    "priority": 0,
    "categoryParent": "<id>",
    "description": "How to get started with your PaceKB Knowledge Base",
    "longDescription": "<h1>Getting started is easy with PaceKB. Here is how you can do it:</h1>",
    "thumbnailUrl": "https://pacekb.com/thumbnail.jpg",
    "iconUrl": "https://pacekb.com/icon.svg",
    "lang": "fr"
}'
```

</RequestExample>

<ResponseExample>

```json Response
{
  "success": true,
  "message": "Category has been updated successfully"
}
```

</ResponseExample>
